% --- Slide 7b: Database Design (Migration Story) ---
\begin{frame}{2. System Architecture}
    \framesubtitle{Database Migration: SQLite $\rightarrow$ Neon Postgres}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Problem:} Split Brain --- Brain and Body could not share a local SQLite file.
        \vspace{0.3cm}

        \textbf{Fix:} Cloud Postgres (Neon) as Single Source of Truth.
        \vspace{0.3cm}

        \textbf{Compatibility Layer:}
        \begin{itemize}
            \item \texttt{PGShimCursor} --- translates \texttt{?} $\rightarrow$ \texttt{\%s} placeholders
            \item Auto-appends \texttt{RETURNING id} for inserts
            \item Zero application-code changes needed
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{Key Tables (13 entities):}
        \begin{itemize}
            \item \texttt{users} --- OAuth, roles, manager hierarchy
            \item \texttt{products} --- inventory + import price
            \item \texttt{import\_transactions} / \texttt{import\_details}
            \item \texttt{export\_transactions} / \texttt{export\_details}
            \item \texttt{sales} --- POS with JSON items
            \item \texttt{workflows} --- JSON DAG definitions
            \item \texttt{se\_automations} --- scheduled tasks
            \item \texttt{ai\_chat\_history} --- agent memory
            \item \texttt{activity\_logs} --- audit trail
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{4. Multi-Agent System}

    \textbf{Engine:} Kahn's Algorithm (Topological Sort) on JSON DAGs [3].
    \vspace{0.4cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{Use Cases:}
        \begin{itemize}
            \item \textbf{UC-01:} Auto-import invoices\\(VLM $\rightarrow$ OCR $\rightarrow$ Validate)
            \item \textbf{UC-02:} Strategic restock\\(Inventory $\rightarrow$ Forecast $\rightarrow$ Email)
            \item \textbf{UC-03:} Chat-based automation\\(Chat $\rightarrow$ CoderAgent $\rightarrow$ Deploy)
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{Pipeline Characteristics:}
        \begin{itemize}
            \item Chat $\rightarrow$ CoderAgent generates JSON DAG
            \item Topological ordering ensures dependency safety
            \item Each node: atomic, idempotent action
            \item Failure at any node $\rightarrow$ mark failed, skip dependent nodes
        \end{itemize}
    \end{columns}
\end{frame}

% --- Slide 13b2: Async Polling Architecture ---
\begin{frame}{5. User Interface}
    \framesubtitle{Asynchronous Polling --- Deep Research Pipeline}
    \begin{columns}
        \column{0.45\textwidth}
        \textbf{Why Polling?}
        \begin{itemize}
            \item AI inference takes $\sim$8--30s (remote GPU)
            \item HTTP request would \textbf{timeout} synchronously
            \item Solution: \textbf{fire-and-poll} pattern
        \end{itemize}
        \vspace{0.3cm}
        \textbf{Backend (Flask):}
        \begin{enumerate}
            \item \texttt{POST /api/ai/chat} $\rightarrow$ spawns \texttt{threading.Thread}
            \item Returns \texttt{job\_id} (UUID) immediately
            \item Thread writes status to \texttt{jobs/\{id\}.json}
            \item \texttt{GET /api/ai/status/\{id\}} returns current state
        \end{enumerate}

        \column{0.55\textwidth}
        \textbf{Frontend Polling (1.5s interval):}
        \begin{enumerate}
            \item Show animated ``Deep Research'' UI card
            \item Step through cognitive phases:
            \begin{itemize}
                \item[\ding{182}] \textbf{Analyzing Intent} (ManagerAgent routing)
                \item[\ding{183}] \textbf{Querying Database} (schema injection)
                \item[\ding{184}] \textbf{Designing Workflow} (CoderAgent)
                \item[\ding{185}] \textbf{Writing Code} (JSON DAG generation)
                \item[\ding{186}] \textbf{Saving Workspace} (DB persist)
            \end{itemize}
            \item On \texttt{status=completed} $\rightarrow$ render response
            \item On \texttt{status=failed} $\rightarrow$ show error
        \end{enumerate}
    \end{columns}
    \vfill
    \centering
    
\end{frame}
